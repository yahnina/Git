git init - создает репозиторий в текущем каталоге
git add ... Добавляет индексирующий файл
git commit -m "Коментарий" фиксация изменений (commit
git config --global user.email ... - задаем глобальные переменные (почту_)
git config --global user.name ... - задаем глобальные переменные (имя_)
git clone <url> - клонируем гит из внешнего хранилища
git clone <url> <folder name>- клонируем гит из внешнего хранилища в указанную нами папку
git status - выводит, какие файлы в каком состоянии находятся 
(неотслеживаемые (незакоммиченые, они добавляются командой add)
 и отслеживаемые (могут быть неизмененные, измененные, индексированные (staged))
git log - выводит список сохраненных коммитов
git log -p показывает разницу между двумя коммитами (например, -2 - между двумя последними)
git remote add <сокрашение> <URL> добавляет удаленный репозиторий, потом вместо URL можно вводить сокращение
git push - помещает репозиторий во внешнее хранилище
git push <server> <branch> помешает репозиторий на сервер в определенную ветку
git pull - получает репозиторий из внешнего хранилища
git remote raname <first name> <second name> переименовывает удаленный репозиторий
git remote rm <name> удаляет удаленный репозиторий
